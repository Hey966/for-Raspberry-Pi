  <script>
    const API_URL = "/attendance/today.json";
    let rawItems = [];
    let currentSortKey = "name";
    let currentSortDir = "asc";
    let lastDataHash = ""; // 用於檢測資料是否有變動

    function fetchAttendance() {
      // 視覺反饋：按下按鈕時稍微變色
      const btn = document.getElementById("refreshBtn");
      btn.innerText = "讀取中...";
      
      fetch(API_URL)
        .then(res => res.json())
        .then(data => {
          const newItems = data.items || [];
          
          // 檢查資料是否有實質變動 (簡單比較長度或內容)
          const currentHash = JSON.stringify(newItems);
          if (currentHash !== lastDataHash) {
            rawItems = newItems;
            lastDataHash = currentHash;
            
            // 只有當「班級篩選」還是「全部」時，才重新生成選項，避免干擾使用者操作
            const classFilter = document.getElementById("classFilter");
            if (classFilter.value === "all") {
              renderClassOptions(rawItems);
            }
            
            renderAll();
          }
          btn.innerText = "重新整理";
          document.getElementById("syncStatus").innerHTML = '<span style="color:#22c55e">●</span> 自動同步中';
        })
        .catch(err => {
          console.error("載入出勤資料失敗:", err);
          btn.innerText = "重試";
          document.getElementById("syncStatus").innerHTML = '<span style="color:#ef4444">●</span> 連線中斷';
        });
    }

    function renderClassOptions(allItems) {
      const sel = document.getElementById("classFilter");
      if (!sel) return;

      const currentVal = sel.value;
      const set = new Set();
      allItems.forEach(it => {
        if (it.class) set.add(it.class);
      });

      const classes = Array.from(set).sort();
      let html = '<option value="all">全部</option>';
      classes.forEach(c => {
        html += `<option value="${c}">${c}</option>`;
      });
      sel.innerHTML = html;
      sel.value = currentVal; // 保持選中項
    }

    function renderAll() {
      const searchValue = document.getElementById("searchInput").value.trim().toLowerCase();
      const statusFilter = document.getElementById("statusFilter").value;
      const classFilter  = document.getElementById("classFilter").value;
      const lateOnly     = document.getElementById("lateOnly").checked;

      let items = rawItems.slice();

      if (searchValue) {
        items = items.filter(it => (it.name || "").toLowerCase().includes(searchValue));
      }
      if (statusFilter !== "all") {
        items = items.filter(it => it.status === statusFilter);
      }
      if (classFilter !== "all") {
        items = items.filter(it => (it.class || "") === classFilter);
      }
      if (lateOnly) {
        items = items.filter(it => it.is_late === true);
      }

      items.sort((a, b) => {
        let va = a[currentSortKey] || "";
        let vb = b[currentSortKey] || "";

        if (currentSortKey === "time_str") {
          if (!va && !vb) return 0;
          if (!va) return 1;
          if (!vb) return -1;
        }

        if (typeof va === "string") va = va.toLowerCase();
        if (typeof vb === "string") vb = vb.toLowerCase();

        if (va < vb) return currentSortDir === "asc" ? -1 : 1;
        if (va > vb) return currentSortDir === "asc" ? 1 : -1;
        return 0;
      });

      renderTableBody(items);
      renderSummary(rawItems);
      renderClassSummary(rawItems);
      renderRowCount(items);
      renderSortIndicators();
    }

    function getInitials(name) {
      if (!name) return "?";
      if (name.length <= 2) return name;
      return name.slice(-2);
    }

    function renderTableBody(items) {
      const tbody = document.getElementById("tbody");
      tbody.innerHTML = "";

      items.forEach(row => {
        const tr = document.createElement("tr");

        const tdName = document.createElement("td");
        const nameWrap = document.createElement("div");
        nameWrap.className = "name-cell";
        const avatar = document.createElement("div");
        avatar.className = "name-avatar";
        avatar.textContent = getInitials(row.name || "");
        const nameText = document.createElement("span");
        nameText.textContent = row.name || "";
        nameWrap.appendChild(avatar);
        nameWrap.appendChild(nameText);
        tdName.appendChild(nameWrap);
        tr.appendChild(tdName);

        const tdClass = document.createElement("td");
        tdClass.textContent = row.class || "未設定";
        tr.appendChild(tdClass);

        const tdStatus = document.createElement("td");
        const statusSpan = document.createElement("span");
        if (row.status === "checked_in") {
          statusSpan.className = "status-pill status-ok";
          statusSpan.innerHTML = '<span class="status-dot-ok"></span><span>已簽到</span>';
        } else if (row.status === "on_leave" || row.has_leave) {
          statusSpan.className = "status-pill status-leave";
          statusSpan.innerHTML = '<span class="status-dot-leave"></span><span>請假（批准）</span>';
        } else {
          statusSpan.className = "status-pill status-miss";
          statusSpan.innerHTML = '<span class="status-dot-miss"></span><span>未簽到</span>';
        }
        tdStatus.appendChild(statusSpan);
        tr.appendChild(tdStatus);

        const tdTime = document.createElement("td");
        tdTime.className = "time-cell";
        if (row.time_str) {
          tdTime.textContent = row.time_str;
        } else {
          tdTime.textContent = "尚未簽到";
          tdTime.classList.add("empty");
        }
        tr.appendChild(tdTime);

        const tdLate = document.createElement("td");
        if (row.status === "checked_in") {
          const lateSpan = document.createElement("span");
          if (row.is_late === true) {
            lateSpan.className = "status-pill status-miss";
            lateSpan.innerHTML = '<span class="status-dot-miss"></span><span>遲到</span>';
          } else {
            lateSpan.className = "status-pill status-ok";
            lateSpan.innerHTML = '<span class="status-dot-ok"></span><span>準時</span>';
          }
          tdLate.appendChild(lateSpan);
        } else if (row.status === "on_leave" || row.has_leave) {
          const leaveSpan = document.createElement("span");
          leaveSpan.className = "status-pill status-leave";
          leaveSpan.innerHTML = '<span class="status-dot-leave"></span><span>請假</span>';
          tdLate.appendChild(leaveSpan);
        } else {
          if (row.is_late === true) {
            const lateSpan = document.createElement("span");
            lateSpan.className = "status-pill status-miss";
            lateSpan.innerHTML = '<span class="status-dot-miss"></span><span>遲到（未簽到）</span>';
            tdLate.appendChild(lateSpan);
          } else {
            tdLate.textContent = "—";
            tdLate.className = "time-cell empty";
          }
        }
        tr.appendChild(tdLate);

        tbody.appendChild(tr);
      });
    }

    function renderClassSummary(allItems) {
      const container = document.getElementById("classSummary");
      if (!container) return;

      const classMap = new Map();
      allItems.forEach(it => {
        const cls = it.class || "未設定";
        if (!classMap.has(cls)) {
          classMap.set(cls, {
            className: cls,
            total: 0,
            checked: 0,
            late: 0,
            absent: 0,
            leave: 0,
          });
        }
        const rec = classMap.get(cls);
        rec.total += 1;

        const isLeave = (it.status === "on_leave" || it.has_leave);
        if (isLeave) {
          rec.leave += 1;
        } else if (it.status === "checked_in") {
          rec.checked += 1;
          if (it.is_late === true) rec.late += 1;
        } else {
          rec.absent += 1;
          if (it.is_late === true) rec.late += 1;
        }
      });

      const rows = Array.from(classMap.values()).sort((a, b) => {
        if (a.className === "未設定") return 1;
        if (b.className === "未設定") return -1;
        return a.className.localeCompare(b.className, "zh-Hant");
      });

      let tableHtml = `
        <div class="class-summary-title">
          <div>班級統計</div>
          <span>點選某一班級，可快速套用下方列表的「班級篩選」。</span>
        </div>
        <table class="class-summary-table">
          <thead>
            <tr>
              <th>班級</th>
              <th>總人數</th>
              <th>已簽到</th>
              <th>遲到</th>
              <th>未簽到</th>
              <th>請假</th>
              <th>出勤率</th>
            </tr>
          </thead>
          <tbody>
      `;

      rows.forEach(rec => {
        const denom = Math.max(0, rec.total - rec.leave);
        const rate = denom === 0 ? 0 : Math.round(rec.checked * 1000 / denom) / 10;
        const rateClass = rate >= 90 ? "class-rate-good" : "class-rate-bad";
        tableHtml += `
          <tr data-class="${rec.className}">
            <td class="class-name-cell">${rec.className}</td>
            <td>${rec.total}</td>
            <td>${rec.checked}</td>
            <td>${rec.late}</td>
            <td>${rec.absent}</td>
            <td>${rec.leave}</td>
            <td class="${rateClass}">${rate}%</td>
          </tr>
        `;
      });

      tableHtml += `
          </tbody>
        </table>
      `;
      container.innerHTML = tableHtml;

      const rowsEl = container.querySelectorAll("tbody tr[data-class]");
      rowsEl.forEach(tr => {
        tr.addEventListener("click", () => {
          const cls = tr.getAttribute("data-class");
          const sel = document.getElementById("classFilter");
          if (!sel) return;
          if (cls === "未設定") sel.value = "all";
          else sel.value = cls;
          renderAll();
        });
      });
    }

    function renderRowCount(items) {
      document.getElementById("rowCountLabel").textContent = `顯示筆數：${items.length}`;
    }

    function renderSortIndicators() {
      const keys = ["name", "class", "status", "time_str"];
      keys.forEach(k => {
        const span = document.getElementById("sortIndicator-" + k);
        if (!span) return;
        if (k === currentSortKey) {
          span.textContent = currentSortDir === "asc" ? "▲" : "▼";
        } else {
          span.textContent = "";
        }
      });
    }

    function renderSummary(allItems) {
      const container = document.getElementById("summary");
      if (!container) return;

      const total   = allItems.length;
      const checked = allItems.filter(it => it.status === "checked_in").length;
      const leave   = allItems.filter(it => it.status === "on_leave" || it.has_leave).length;
      const late    = allItems.filter(it => it.status === "checked_in" && it.is_late === true).length;
      const absent  = total - checked - leave;
      const denom   = Math.max(0, total - leave);
      const rate    = denom === 0 ? 0 : Math.round(checked * 1000 / denom) / 10;

      container.innerHTML = `
        <div class="summary-card2">
          <div class="summary-label2">總人數</div>
          <div class="summary-value2">${total}</div>
        </div>
        <div class="summary-card2">
          <div class="summary-label2">已簽到 <span class="summary-tag-ok">正常</span></div>
          <div class="summary-value2">${checked}</div>
        </div>
        <div class="summary-card2">
          <div class="summary-label2">未簽到 / 遲到 / 請假</div>
          <div class="summary-value2">${absent} / ${late} / ${leave}</div>
        </div>
        <div class="summary-card2">
          <div class="summary-label2">出勤率 <span class="summary-tag-rate">已扣除請假</span></div>
          <div class="summary-value2">${rate}%</div>
        </div>
      `;
    }

    function setupEvents() {
      document.getElementById("searchInput").addEventListener("input", renderAll);
      document.getElementById("statusFilter").addEventListener("change", renderAll);
      document.getElementById("classFilter").addEventListener("change", renderAll);
      document.getElementById("lateOnly").addEventListener("change", renderAll);
      document.getElementById("refreshBtn").addEventListener("click", fetchAttendance);

      const ths = document.querySelectorAll("th.sortable");
      ths.forEach(th => {
        th.addEventListener("click", () => {
          const key = th.getAttribute("data-key");
          if (key === currentSortKey) {
            currentSortDir = (currentSortDir === "asc") ? "desc" : "asc";
          } else {
            currentSortKey = key;
            currentSortDir = "asc";
          }
          renderAll();
        });
      });
    }

    setupEvents();
    fetchAttendance();

    // 每 5 秒自動抓取一次新資料，實現「即時監控」效果
    setInterval(fetchAttendance, 5000);
  </script>
